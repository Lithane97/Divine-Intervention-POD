# animal lifestyle
DI_add_animal_pp = {
	scope = character
	
	effect = {
		if = {
			limit = {
				has_variable = DI_pinned_char_var
			}
			every_living_character = {
				limit = { has_character_flag = pinned_for_edit }

				hidden_effect = {
					
					add_animal_lifestyle_perk_points = 1
				}
			}
		}
		else = {
			hidden_effect = {
				add_animal_lifestyle_perk_points = 1
			}
		}
	}
}
