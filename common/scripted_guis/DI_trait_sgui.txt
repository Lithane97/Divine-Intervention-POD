#Lifestyle Reveler
#Add
DI_add_lifestyle_reveler_xp = {
	scope = character

	effect = {
		if = {
			limit = {
				has_variable = DI_marked_char_var
			}
			every_living_character = {
				limit = { has_character_flag = marked_for_edit }
				hidden_effect = {
					if = {
						limit = {
							has_trait = lifestyle_reveler
						}
						add_trait_xp = {
							trait = lifestyle_reveler
							value = 50
						}
					}
					else = {
						add_trait = lifestyle_reveler
					}
				}
			}
		}
		else = {
			hidden_effect = {
				if = {
					limit = {
						has_trait = lifestyle_reveler
					}
					add_trait_xp = {
						trait = lifestyle_reveler
						value = 50
					}
				}
				else = {
					add_trait = lifestyle_reveler
				}
			}
		}
	}
}
